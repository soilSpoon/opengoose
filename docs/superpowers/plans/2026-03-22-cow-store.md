> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add in-memory Copy-on-Write branching to Board's WorkItem storage, using `Arc<BTreeMap>` with CRDT merge and SHA-256 commit log, persisted to SQLite on merge.

**Architecture:** New files `merge.rs`, `store.rs`, `branch.rs` in opengoose-board. Board struct gains a `CowStore` field. Worker creates explicit `Branch` handles. Stamps/Relations stay in SQLite unchanged. Tasks are ordered by dependency: CRDT types → store/branch → persistence → Board integration → Worker integration.

**Tech Stack:** Rust, `Arc<BTreeMap>`, `sha2` crate, SeaORM, SQLite WAL

**Spec:** `docs/superpowers/specs/2026-03-22-cow-store-design.md`

**Naming convention:** Do NOT use `_` prefix/suffix for variable names. Fix existing violations in files you touch.

use chrono::{DateTime, Utc};
use std::collections::BTreeSet;

use crate::work_item::Priority;

// ── Trait ──────────────────────────────────────────────

/// Conflict-free merge of two diverged values.
///
/// Implementations must satisfy:
/// - Commutativity: a.merge(b) == b.merge(a)
/// - Associativity: a.merge(b.merge(c)) == a.merge(b).merge(c)
/// - Idempotency:   a.merge(a) == a
pub trait Mergeable {
    fn merge(&self, other: &Self) -> Self;
}

// ── LWW Register ──────────────────────────────────────

#[derive(Debug, Clone, PartialEq, Eq)]
pub struct LwwField<T> {
    pub value: T,
    pub updated_at: DateTime<Utc>,
}

impl<T: Clone> Mergeable for LwwField<T> {
    fn merge(&self, other: &Self) -> Self {
        if self.updated_at >= other.updated_at {
            self.clone()
        } else {
            other.clone()
        }
    }
}

// ── Priority: Max-register ────────────────────────────

impl Mergeable for Priority {
    fn merge(&self, other: &Self) -> Self {
        std::cmp::max(*self, *other)
    }
}

// ── Tags: G-Set (grow-only union) ─────────────────────

pub fn merge_tags(a: &[String], b: &[String]) -> Vec<String> {
    let mut union: BTreeSet<&str> = BTreeSet::new();
    for t in a.iter().chain(b.iter()) {
        union.insert(t.as_str());
    }
    union.into_iter().map(String::from).collect()
}

// ── Result types ──────────────────────────────────────

pub struct MergeResult {
    pub merged_items: Vec<MergedItem>,
    pub commit_id: u64,
}

pub struct MergedItem {
    pub item_id: i64,
    pub item: WorkItem,
    pub convergences: Vec<Convergence>,
}

pub struct Convergence {
    pub field: &'static str,
    pub branch_value: String,
    pub main_value: String,
    pub converged_to: String,
    pub strategy: MergeStrategy,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum MergeStrategy {
    OneSided,
    MaxRegister,
    GrowSet,
    LastWriteWins,
}
```

use crate::work_item::{RigId, Status, WorkItem};

pub fn merge_work_item(base: &WorkItem, branch: &WorkItem, main: &WorkItem) -> MergedItem {
    let mut convergences = Vec::new();

    // Status: LWW (uses updated_at because status can go backward)
    let status = merge_lww(
        "status",
        &base.status, &branch.status, &main.status,
        base.updated_at, branch.updated_at, main.updated_at,
        |s| format!("{s:?}"),
        &mut convergences,
    );

    // Priority: Max-register
    let priority = merge_max(
        "priority",
        &base.priority, &branch.priority, &main.priority,
        &mut convergences,
    );

    // Tags: G-Set union
    let tags = merge_grow_set(
        "tags",
        &base.tags, &branch.tags, &main.tags,
        &mut convergences,
    );

    // claimed_by: LWW
    let claimed_by = merge_lww(
        "claimed_by",
        &base.claimed_by, &branch.claimed_by, &main.claimed_by,
        base.updated_at, branch.updated_at, main.updated_at,
        |c| format!("{c:?}"),
        &mut convergences,
    );

    let merged_item = WorkItem {
        id: base.id,
        title: base.title.clone(),
        description: base.description.clone(),
        created_by: base.created_by.clone(),
        created_at: base.created_at,
        status,
        priority,
        tags,
        claimed_by,
        updated_at: std::cmp::max(branch.updated_at, main.updated_at),
    };

    MergedItem {
        item_id: base.id,
        item: merged_item,
        convergences,
    }
}

// ── 3-way merge helpers ───────────────────────────────

fn merge_lww<T: Clone + PartialEq>(
    field: &'static str,
    base: &T, branch: &T, main: &T,
    base_ts: DateTime<Utc>, branch_ts: DateTime<Utc>, main_ts: DateTime<Utc>,
    fmt: impl Fn(&T) -> String,
    convergences: &mut Vec<Convergence>,
) -> T {
    let branch_changed = branch != base;
    let main_changed = main != base;

    match (branch_changed, main_changed) {
        (false, false) => base.clone(),
        (true, false) => {
            convergences.push(Convergence {
                field,
                branch_value: fmt(branch),
                main_value: fmt(main),
                converged_to: fmt(branch),
                strategy: MergeStrategy::OneSided,
            });
            branch.clone()
        }
        (false, true) => {
            convergences.push(Convergence {
                field,
                branch_value: fmt(branch),
                main_value: fmt(main),
                converged_to: fmt(main),
                strategy: MergeStrategy::OneSided,
            });
            main.clone()
        }
        (true, true) => {
            // Both changed — LWW by updated_at
            let winner = if branch_ts >= main_ts { branch } else { main };
            convergences.push(Convergence {
                field,
                branch_value: fmt(branch),
                main_value: fmt(main),
                converged_to: fmt(winner),
                strategy: MergeStrategy::LastWriteWins,
            });
            winner.clone()
        }
    }
}

fn merge_max<T: Clone + PartialEq + Ord>(
    field: &'static str,
    base: &T, branch: &T, main: &T,
    convergences: &mut Vec<Convergence>,
) -> T
where T: std::fmt::Debug
{
    let branch_changed = branch != base;
    let main_changed = main != base;

    match (branch_changed, main_changed) {
        (false, false) => base.clone(),
        (true, false) => {
            convergences.push(Convergence {
                field,
                branch_value: format!("{branch:?}"),
                main_value: format!("{main:?}"),
                converged_to: format!("{branch:?}"),
                strategy: MergeStrategy::OneSided,
            });
            branch.clone()
        }
        (false, true) => {
            convergences.push(Convergence {
                field,
                branch_value: format!("{branch:?}"),
                main_value: format!("{main:?}"),
                converged_to: format!("{main:?}"),
                strategy: MergeStrategy::OneSided,
            });
            main.clone()
        }
        (true, true) => {
            let winner = std::cmp::max(branch, main);
            convergences.push(Convergence {
                field,
                branch_value: format!("{branch:?}"),
                main_value: format!("{main:?}"),
                converged_to: format!("{winner:?}"),
                strategy: MergeStrategy::MaxRegister,
            });
            winner.clone()
        }
    }
}

fn merge_grow_set(
    field: &'static str,
    base: &[String], branch: &[String], main: &[String],
    convergences: &mut Vec<Convergence>,
) -> Vec<String> {
    let branch_changed = branch != base;
    let main_changed = main != base;

    match (branch_changed, main_changed) {
        (false, false) => base.to_vec(),
        (true, false) => {
            convergences.push(Convergence {
                field,
                branch_value: format!("{branch:?}"),
                main_value: format!("{main:?}"),
                converged_to: format!("{branch:?}"),
                strategy: MergeStrategy::OneSided,
            });
            branch.to_vec()
        }
        (false, true) => {
            convergences.push(Convergence {
                field,
                branch_value: format!("{branch:?}"),
                main_value: format!("{main:?}"),
                converged_to: format!("{main:?}"),
                strategy: MergeStrategy::OneSided,
            });
            main.to_vec()
        }
        (true, true) => {
            let merged = merge_tags(branch, main);
            convergences.push(Convergence {
                field,
                branch_value: format!("{branch:?}"),
                main_value: format!("{main:?}"),
                converged_to: format!("{merged:?}"),
                strategy: MergeStrategy::GrowSet,
            });
            merged
        }
    }
}

pub struct Branch {
    pub(crate) name: RigId,
    pub(crate) data: Arc<BTreeMap<i64, WorkItem>>,
    pub(crate) base_data: Arc<BTreeMap<i64, WorkItem>>,  // unmodified snapshot for 3-way merge
    pub(crate) base_commit: u64,
}

impl Branch {
    pub(crate) fn new(name: RigId, data: Arc<BTreeMap<i64, WorkItem>>, base_commit: u64) -> Self {
        let base_data = Arc::clone(&data); // both point to same data initially; CoW separates on write
        Self { name, data, base_data, base_commit }
    }

    pub fn name(&self) -> &RigId {
        &self.name
    }

    pub fn get(&self, id: i64) -> Option<&WorkItem> {
        self.data.get(&id)
    }

    pub fn list(&self) -> impl Iterator<Item = &WorkItem> {
        self.data.values()
    }

    pub fn ready(&self, blocked_ids: &HashSet<i64>) -> Vec<&WorkItem> {
        let mut items: Vec<&WorkItem> = self.data.values()
            .filter(|item| item.status == Status::Open && !blocked_ids.contains(&item.id))
            .collect();
        items.sort_by(|a, b| b.priority.cmp(&a.priority));
        items
    }

    pub fn insert(&mut self, item: WorkItem) {
        Arc::make_mut(&mut self.data).insert(item.id, item);
    }

    pub fn update(&mut self, id: i64, f: impl FnOnce(&mut WorkItem)) {
        if let Some(item) = Arc::make_mut(&mut self.data).get_mut(&id) {
            f(item);
        }
    }

    pub fn remove(&mut self, id: i64) {
        Arc::make_mut(&mut self.data).remove(&id);
    }
}

use std::collections::BTreeMap;
use std::sync::Arc;

use chrono::{DateTime, Utc};
use sha2::{Digest, Sha256};

use crate::branch::Branch;
use crate::merge::{merge_work_item, MergeResult, MergedItem};
use crate::work_item::{BoardError, RigId, WorkItem};

// ── Commit types ──────────────────────────────────────

#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
pub struct CommitId(pub u64);

#[derive(Debug, Clone)]
pub struct Commit {
    pub id: CommitId,
    pub parent: Option<CommitId>,
    pub root_hash: [u8; 32],
    pub branch: RigId,
    pub message: String,
    pub timestamp: DateTime<Utc>,
}

// ── CowStore ──────────────────────────────────────────

pub struct CowStore {
    pub(crate) main: Arc<BTreeMap<i64, WorkItem>>,
    pub(crate) commits: Vec<Commit>,
    next_commit_id: u64,
}

impl CowStore {
    pub fn new() -> Self {
        Self {
            main: Arc::new(BTreeMap::new()),
            commits: Vec::new(),
            next_commit_id: 0,
        }
    }

    pub fn from_items(items: BTreeMap<i64, WorkItem>, commits: Vec<Commit>) -> Self {
        let next_commit_id = commits.last().map(|c| c.id.0 + 1).unwrap_or(0);
        Self {
            main: Arc::new(items),
            commits,
            next_commit_id,
        }
    }

    /// Insert an item directly into main (used by post(), which assigns ID via SQLite).
    pub fn insert_to_main(&mut self, item: WorkItem) {
        Arc::make_mut(&mut self.main).insert(item.id, item);
    }

    /// Update an item directly in main (used for non-branched operations like retry, abandon).
    pub fn update_in_main(&mut self, id: i64, f: impl FnOnce(&mut WorkItem)) {
        if let Some(item) = Arc::make_mut(&mut self.main).get_mut(&id) {
            f(item);
        }
    }

    /// Remove an item from main.
    pub fn remove_from_main(&mut self, id: i64) {
        Arc::make_mut(&mut self.main).remove(&id);
    }

    /// Create a snapshot branch. O(1) via Arc::clone.
    pub fn branch(&mut self, rig_id: &RigId) -> Branch {
        let base_commit = self.commits.last().map(|c| c.id.0).unwrap_or(0);
        Branch::new(rig_id.clone(), Arc::clone(&self.main), base_commit)
    }

    /// 3-way merge: base (branch creation snapshot) vs branch vs current main.
    /// Uses branch.base_data as the common ancestor for true 3-way comparison.
    pub fn merge(&mut self, branch: Branch) -> Result<MergeResult, BoardError> {
        let branch_name = branch.name.clone();
        let mut merged_items = Vec::new();
        let main = Arc::make_mut(&mut self.main);

        // For each item in the branch snapshot
        for (id, branch_item) in branch.data.iter() {
            let base_item = branch.base_data.get(id);
            let main_item = main.get(id);

            match (base_item, main_item) {
                // Item existed at branch creation, exists in both branch and main
                (Some(base), Some(current_main)) => {
                    let branch_changed = branch_item != base;
                    let main_changed = current_main != base;

                    match (branch_changed, main_changed) {
                        (false, _) => {} // Branch didn't touch it — keep main as-is
                        (true, false) => {
                            // Only branch changed — take branch version
                            main.insert(*id, branch_item.clone());
                            merged_items.push(MergedItem {
                                item_id: *id,
                                item: branch_item.clone(),
                                convergences: vec![],
                            });
                        }
                        (true, true) => {
                            // Both changed — CRDT 3-way merge
                            let merged = merge_work_item(base, branch_item, current_main);
                            main.insert(*id, merged.item.clone());
                            merged_items.push(merged);
                        }
                    }
                }
                // Item was added by branch (not in base, not in main)
                (None, None) => {
                    main.insert(*id, branch_item.clone());
                    merged_items.push(MergedItem {
                        item_id: *id,
                        item: branch_item.clone(),
                        convergences: vec![],
                    });
                }
                // Item was added by branch but also appeared in main independently
                (None, Some(current_main)) => {
                    let merged = merge_work_item(branch_item, branch_item, current_main);
                    main.insert(*id, merged.item.clone());
                    merged_items.push(merged);
                }
                // Item existed in base but was removed from main — branch changes lost
                (Some(_), None) => {}
            }
        }

        // Items removed by branch: existed in base_data but not in branch.data
        for id in branch.base_data.keys() {
            if !branch.data.contains_key(id) {
                // Branch explicitly removed this item
                main.remove(id);
            }
        }

        drop(main);

        let commit = self.append_commit(&branch_name, format!("merge {branch_name}"));
        let commit_id = commit.id.0;

        Ok(MergeResult {
            merged_items,
            commit_id,
        })
    }

    /// Discard a branch without merging.
    pub fn discard(&mut self, branch: Branch) {
        // Branch is dropped, its Arc data ref-count decreases.
        // If main wasn't modified through the branch, nothing happens.
        let _ = branch;
    }

    /// Read-only access to main.
    pub fn main_snapshot(&self) -> Arc<BTreeMap<i64, WorkItem>> {
        Arc::clone(&self.main)
    }

    /// Get an item from main.
    pub fn get(&self, id: i64) -> Option<&WorkItem> {
        self.main.get(&id)
    }

    /// List all items in main.
    pub fn list_main(&self) -> impl Iterator<Item = &WorkItem> {
        self.main.values()
    }

    /// Current commits.
    pub fn commits(&self) -> &[Commit] {
        &self.commits
    }

    // ── Internal ──────────────────────────────────────

    fn compute_root_hash(data: &BTreeMap<i64, WorkItem>) -> [u8; 32] {
        let mut hasher = Sha256::new();
        for (id, item) in data.iter() {
            hasher.update(id.to_le_bytes());
            // Hash all fields for correctness
            hasher.update(serde_json::to_vec(item).unwrap_or_default());
        }
        hasher.finalize().into()
    }

    fn append_commit(&mut self, branch: &RigId, message: String) -> Commit {
        let root_hash = Self::compute_root_hash(&self.main);
        let parent = self.commits.last().map(|c| c.id);
        let id = CommitId(self.next_commit_id);
        self.next_commit_id += 1;
        let commit = Commit {
            id,
            parent,
            root_hash,
            branch: branch.clone(),
            message,
            timestamp: Utc::now(),
        };
        self.commits.push(commit.clone());
        commit
    }
}

use sea_orm::{
    ActiveModelTrait, ConnectionTrait, DatabaseConnection, EntityTrait, QueryOrder, Set,
    TransactionTrait,
};

impl CowStore {
    /// Persist current main state and commit log to SQLite.
    pub async fn persist(&self, db: &DatabaseConnection) -> Result<(), BoardError> {
        use crate::entity;

        let txn = db.begin().await.map_err(crate::board::db_err)?;

        // Upsert all work items from main using SeaORM (parameterized, no SQL injection)
        for item in self.main.values() {
            let tags_json = serde_json::to_string(&item.tags).unwrap_or_default();
            let active = entity::work_item::ActiveModel {
                id: Set(item.id),
                title: Set(item.title.clone()),
                description: Set(item.description.clone()),
                status: Set(item.status),
                priority: Set(item.priority),
                tags: Set(Some(tags_json)),
                created_by: Set(item.created_by.0.clone()),
                claimed_by: Set(item.claimed_by.as_ref().map(|r| r.0.clone())),
                created_at: Set(item.created_at),
                updated_at: Set(item.updated_at),
            };
            entity::work_item::Entity::insert(active)
                .on_conflict(
                    sea_orm::sea_query::OnConflict::column(entity::work_item::Column::Id)
                        .update_columns([
                            entity::work_item::Column::Title,
                            entity::work_item::Column::Description,
                            entity::work_item::Column::Status,
                            entity::work_item::Column::Priority,
                            entity::work_item::Column::Tags,
                            entity::work_item::Column::ClaimedBy,
                            entity::work_item::Column::UpdatedAt,
                        ])
                        .to_owned(),
                )
                .exec(&txn)
                .await
                .map_err(crate::board::db_err)?;
        }

        // Persist commits
        for commit in &self.commits {
            let exists: Vec<entity::commit_log::Model> = entity::commit_log::Entity::find_by_id(commit.id.0)
                .all(&txn)
                .await
                .map_err(crate::board::db_err)?;
            if exists.is_empty() {
                let active = entity::commit_log::ActiveModel {
                    id: Set(commit.id.0),
                    parent_id: Set(commit.parent.map(|p| p.0)),
                    root_hash: Set(commit.root_hash.to_vec()),
                    branch: Set(commit.branch.0.clone()),
                    message: Set(commit.message.clone()),
                    created_at: Set(commit.timestamp),
                };
                active.insert(&txn).await.map_err(crate::board::db_err)?;
            }
        }

        txn.commit().await.map_err(crate::board::db_err)?;
        Ok(())
    }

    /// Restore CowStore state from SQLite.
    pub async fn restore(db: &DatabaseConnection) -> Result<Self, BoardError> {
        use crate::entity;

        // Load all work items
        let models = entity::work_item::Entity::find()
            .all(db)
            .await
            .map_err(crate::board::db_err)?;

        let mut items = BTreeMap::new();
        for model in models {
            let item = crate::work_item::WorkItem::from(model);
            items.insert(item.id, item);
        }

        // Load commit log
        let commit_models = entity::commit_log::Entity::find()
            .order_by_asc(entity::commit_log::Column::Id)
            .all(db)
            .await
            .map_err(crate::board::db_err)?;

        let commits: Vec<Commit> = commit_models
            .into_iter()
            .map(|m| Commit {
                id: CommitId(m.id as u64),
                parent: m.parent_id.map(|p| CommitId(p as u64)),
                root_hash: {
                    let mut hash = [0u8; 32];
                    let len = m.root_hash.len().min(32);
                    hash[..len].copy_from_slice(&m.root_hash[..len]);
                    hash
                },
                branch: RigId::new(m.branch),
                message: m.message,
                timestamp: m.created_at,
            })
            .collect();

        Ok(Self::from_items(items, commits))
    }
}

async fn try_claim_and_execute(&self) -> anyhow::Result<bool> {
    let board = self.board.as_ref().expect("Worker must have a board");

    // Compute blocked IDs from SQLite (before branching)
    let blocked_ids = board.get_blocked_ids().await?;

    // Create branch (async due to tokio::sync::Mutex)
    let mut branch = board.branch(&self.id).await;

    // Find ready items in branch
    let ready_item_id = {
        let ready = branch.ready(&blocked_ids);
        match ready.first() {
            Some(item) => item.id,
            None => return Ok(false),
        }
    };

    // Claim through branch
    branch.update(ready_item_id, |item| {
        item.status = crate::work_item::Status::Claimed;
        item.claimed_by = Some(self.id.clone());
        item.updated_at = chrono::Utc::now();
    });

    // Merge the claim to main so other systems see it
    board.merge(branch).await?;

    // Now process the item through SQLite as before
    let item = board.get(ready_item_id).await?.ok_or_else(|| {
        anyhow::anyhow!("work item {ready_item_id} disappeared after merge")
    })?;

    info!(rig = %self.id, item_id = item.id, title = %item.title, "claimed work item");

    self.process_claimed_item(&item, board).await;
    Ok(true)
}
```

Note: We claim through Branch for CoW safety, then merge immediately so SQLite stays in sync. The actual `process_claimed_item` still uses `board.submit()` / `board.abandon()` which go through SQLite directly.
